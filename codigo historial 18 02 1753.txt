import { Trash2, Search, ArrowLeft, FileText, Loader2, Calendar, Download, LineChart, Target, Clock, Activity, AlertTriangle, Stethoscope, Scale } from "lucide-react";

const app = initializeApp(firebaseConfig);
const auth = getAuth(app);
const db = getFirestore(app);

// --- DICCIONARIOS ---
const ESTRATEGIAS_LABELS: Record<string, string> = {
    equilibrada: "‚öñÔ∏è Equilibrada", mediterranea: "üçÖ Mediterr√°nea", lowcarb: "ü•© Low Carb",
    atleta: "‚ö° Alto Rendimiento", definicion: "üî• Definici√≥n", bariatrica: "üíß Bari√°trica"
};

const RESTRICCIONES_LABELS: Record<string, string> = {
    normal: "Sin Restricciones", celiaco: "Celiaqu√≠a", lactosa: "Intolerancia Lactosa",
    vegetariano: "Vegetariano", vegano: "Vegano", fodmap: "FODMAP / SIBO",
    menopausia: "Menopausia", diabetes: "Diabetes / RI", hipertension: "Hipertensi√≥n",
    colesterol: "Dislipidemia", reflujo: "Reflujo / Gastritis", gota: "Gota"
};

const OBJETIVOS_READABLE: Record<string, string> = {
    bajar: "P√©rdida de Grasa", mantener: "Mantenci√≥n", subir: "Aumento Muscular"
};

export default function HistorialPage() {
    const router = useRouter(); 
    const [user, setUser] = useState<any>(null);
    const [pautas, setPautas] = useState<any[]>([]);
    const [busqueda, setBusqueda] = useState("");
    const [cargando, setCargando] = useState(true);

    // --- ESTADOS PARA KPIs (M√©tricas Avanzadas) ---
    const [stats, setStats] = useState({
        pacientesMes: 0,
        topObjetivo: "General",
        porcentajeNicho: 0,
        diasUltimaConsulta: 0,
        // Nuevas M√©tricas
        pacientesRiesgo: 0, // Inactivos > 30 d√≠as
        topPatologia: "Ninguna",
        promedioIMC: "0.0"
    });

    // 1. VERIFICAR USUARIO Y CARGAR DATOS
    useEffect(() => {
        const unsubscribe = onAuthStateChanged(auth, (usuario) => {
            if (usuario) {
                setUser(usuario);
                cargarMisDatos(usuario.email);
            } else {
                window.location.href = "/";
            }
        });
        return () => unsubscribe();
    }, []);

    const cargarMisDatos = async (email: string | null) => {
        if (!email) return;
        setCargando(true);
        try {
            const q = query(collection(db, "pautas"), orderBy("fecha", "desc"));
            const querySnapshot = await getDocs(q);
            
            const datos = querySnapshot.docs
                .map(doc => ({ id: doc.id, ...doc.data() }))
                .filter((doc: any) => doc.creadoPor === email || !doc.creadoPor); 

            setPautas(datos);
            calcularMetricas(datos); 

        } catch (error) {
            console.error("Error cargando historial:", error);
        }
        setCargando(false);
    };

    // --- L√ìGICA DE INTELIGENCIA DE NEGOCIO (KPIs) ---
    const calcularMetricas = (data: any[]) => {
        if (data.length === 0) return;

        const now = new Date();
        
        // 1. Pacientes este mes
        const esteMes = data.filter(p => {
            const d = new Date(p.fecha);
            return d.getMonth() === now.getMonth() && d.getFullYear() === now.getFullYear();
        }).length;

        // 2. Nicho Principal (Objetivo)
        const conteoObjetivos: Record<string, number> = {};
        const conteoPatologias: Record<string, number> = {};
        let sumaIMC = 0;
        let countIMC = 0;
        let inactivos = 0;

        data.forEach(p => {
            // Objetivo
            const obj = p.objetivo || "bajar";
            conteoObjetivos[obj] = (conteoObjetivos[obj] || 0) + 1;

            // Patolog√≠a
            const pat = p.restriccion || "normal";
            if(pat !== 'normal') conteoPatologias[pat] = (conteoPatologias[pat] || 0) + 1;

            // Promedio IMC
            if(p.peso && p.altura) {
                const imc = Number(p.peso) / ((Number(p.altura)/100)**2);
                if(imc > 10 && imc < 60) { sumaIMC += imc; countIMC++; }
            }

            // Inactivos > 30 d√≠as
            const fechaPauta = new Date(p.fecha);
            const diffTime = Math.abs(now.getTime() - fechaPauta.getTime());
            const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24)); 
            if(diffDays > 30) inactivos++;
        });
        
        // Ganador Objetivo
        let topObj = "bajar"; let maxCount = 0;
        Object.entries(conteoObjetivos).forEach(([key, val]) => { if (val > maxCount) { maxCount = val; topObj = key; } });
        const porcentaje = Math.round((maxCount / data.length) * 100);

        // Ganador Patolog√≠a
        let topPat = "Sano / Normal"; let maxPat = 0;
        Object.entries(conteoPatologias).forEach(([key, val]) => { if (val > maxPat) { maxPat = val; topPat = key; } });
        if(maxPat === 0) topPat = "Sin Patolog√≠as";
        else topPat = RESTRICCIONES_LABELS[topPat] || topPat;

        // Ultima consulta
        const ultimaFecha = new Date(data[0].fecha);
        const diasUltima = Math.floor((now.getTime() - ultimaFecha.getTime()) / (1000 * 3600 * 24));

        setStats({
            pacientesMes: esteMes,
            topObjetivo: OBJETIVOS_READABLE[topObj] || "General",
            porcentajeNicho: porcentaje,
            diasUltimaConsulta: diasUltima,
            pacientesRiesgo: inactivos,
            topPatologia: topPat,
            promedioIMC: countIMC > 0 ? (sumaIMC / countIMC).toFixed(1) : "0.0"
        });
    };

    // 2. EXPORTAR A EXCEL
    const exportarExcel = () => {
        if (pautas.length === 0) return alert("No hay datos para exportar.");
        let csvContent = "\uFEFFsep=;\nFecha;Hora;Paciente;Estrategia;Peso (kg);Objetivo;Calorias Meta\n";
        pautas.forEach(p => {
            const f = new Date(p.fecha);
            const row = `${f.toLocaleDateString('es-CL')};${f.toLocaleTimeString('es-CL')};${p.paciente};${p.estrategia};${p.peso};${p.objetivo};${p.caloriasMeta}`;
            csvContent += row + "\n";
        });
        const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });
        const url = URL.createObjectURL(blob);
        const link = document.createElement("a");
        link.setAttribute("href", url);
        link.setAttribute("download", `Pacientes_${new Date().toISOString().slice(0,10)}.csv`);
        document.body.appendChild(link);
        link.click();
        document.body.removeChild(link);
    };

    const borrarPauta = async (id: string) => {
        if(!confirm("¬øEliminar este registro permanentemente?")) return;
        try {
            await deleteDoc(doc(db, "pautas", id));
            const nuevaLista = pautas.filter(p => p.id !== id);
            setPautas(nuevaLista);
            calcularMetricas(nuevaLista); 
        } catch (error) { alert("Error al borrar"); }
    };

    const cargarEnCalculadora = (pauta: any) => {
        localStorage.setItem("pauta_editar", JSON.stringify(pauta));
        router.push("/");
    };

    const getObjetivoBadge = (obj: string) => {
        if (obj === 'bajar') return <span className="bg-red-100 text-red-700 px-2 py-1 rounded-md text-[10px] font-black border border-red-200">üìâ D√âFICIT</span>;
        if (obj === 'subir') return <span className="bg-blue-100 text-blue-700 px-2 py-1 rounded-md text-[10px] font-black border border-blue-200">üí™ SUPER√ÅVIT</span>;
        return <span className="bg-green-100 text-green-700 px-2 py-1 rounded-md text-[10px] font-black border border-green-200">‚öì MANTENCI√ìN</span>;
    };

    const pautasFiltradas = pautas.filter(p => p.paciente.toLowerCase().includes(busqueda.toLowerCase()));

    if (cargando) return <div className="min-h-screen flex items-center justify-center text-blue-600 font-bold gap-2 bg-slate-50"><Loader2 className="animate-spin" /> Cargando panel...</div>;

    return (
        <div className="min-h-screen bg-slate-50 p-6 font-sans">
            <div className="max-w-6xl mx-auto">
                
                {/* CABECERA */}
                <div className="flex flex-col md:flex-row justify-between items-center mb-8 gap-4">
                    <div className="flex-1">
                        <Link href="/" className="text-slate-400 hover:text-blue-600 flex items-center gap-2 mb-2 text-xs font-bold transition uppercase tracking-widest">
                            <ArrowLeft className="w-3 h-3" /> Volver al Calculador
                        </Link>
                        <h1 className="text-3xl font-black text-slate-800 tracking-tight">Panel de Control Cl√≠nico</h1>
                        <p className="text-slate-500 text-xs mt-1">
                            Bienvenido, <span className="font-bold text-blue-600">{user?.email}</span>
                        </p>
                    </div>
                    <button onClick={exportarExcel} className="bg-green-600 hover:bg-green-700 text-white px-5 py-3 rounded-xl shadow-lg shadow-green-100 font-bold text-xs flex items-center gap-2 transition-all active:scale-95">
                        <Download className="w-4 h-4" /> Descargar Excel
                    </button>
                </div>

                {/* --- DASHBOARD DE KPIs (6 TARJETAS EN 2 FILAS) --- */}
                <div className="grid grid-cols-1 md:grid-cols-3 gap-4 mb-8">
                    
                    {/* 1. ACTIVIDAD */}
                    <div className="bg-white p-5 rounded-2xl shadow-sm border border-slate-100 flex items-center gap-4 hover:shadow-md transition">
                        <div className="p-3 bg-blue-50 rounded-xl text-blue-600"><Activity className="w-6 h-6" /></div>
                        <div>
                            <p className="text-[10px] font-bold text-slate-400 uppercase tracking-widest">Consultas Este Mes</p>
                            <h3 className="text-2xl font-black text-slate-800">{stats.pacientesMes}</h3>
                        </div>
                    </div>

                    {/* 2. NICHO */}
                    <div className="bg-white p-5 rounded-2xl shadow-sm border border-slate-100 flex items-center gap-4 hover:shadow-md transition">
                        <div className="p-3 bg-purple-50 rounded-xl text-purple-600"><Target className="w-6 h-6" /></div>
                        <div>
                            <p className="text-[10px] font-bold text-slate-400 uppercase tracking-widest">Objetivo Principal</p>
                            <h3 className="text-lg font-black text-slate-800 leading-none">{stats.topObjetivo}</h3>
                            <p className="text-[9px] font-bold text-purple-500 mt-1">{stats.porcentajeNicho}% de tus casos</p>
                        </div>
                    </div>

                    {/* 3. RECENCIA */}
                    <div className="bg-white p-5 rounded-2xl shadow-sm border border-slate-100 flex items-center gap-4 hover:shadow-md transition">
                        <div className={`p-3 rounded-xl ${stats.diasUltimaConsulta > 7 ? 'bg-orange-50 text-orange-500' : 'bg-green-50 text-green-600'}`}><Clock className="w-6 h-6" /></div>
                        <div>
                            <p className="text-[10px] font-bold text-slate-400 uppercase tracking-widest">√öltima Actividad</p>
                            <h3 className="text-xl font-black text-slate-800">{stats.diasUltimaConsulta === 0 ? "Hoy" : `Hace ${stats.diasUltimaConsulta} d√≠as`}</h3>
                        </div>
                    </div>

                    {/* --- NUEVA FILA: INTELIGENCIA DE NEGOCIO --- */}

                    {/* 4. ALERTA DE FUGA (RETENCI√ìN) */}
                    <div className="bg-white p-5 rounded-2xl shadow-sm border border-slate-100 flex items-center gap-4 hover:shadow-md transition border-l-4 border-l-red-400">
                        <div className="p-3 bg-red-50 rounded-xl text-red-500"><AlertTriangle className="w-6 h-6" /></div>
                        <div>
                            <p className="text-[10px] font-bold text-slate-400 uppercase tracking-widest">Inactivos (+30 d√≠as)</p>
                            <h3 className="text-2xl font-black text-slate-800">{stats.pacientesRiesgo} <span className="text-sm font-normal text-slate-400">Pacientes</span></h3>
                            <p className="text-[9px] text-red-400 font-bold">¬°Riesgo de abandono!</p>
                        </div>
                    </div>

                    {/* 5. TOP PATOLOG√çA */}
                    <div className="bg-white p-5 rounded-2xl shadow-sm border border-slate-100 flex items-center gap-4 hover:shadow-md transition">
                        <div className="p-3 bg-teal-50 rounded-xl text-teal-600"><Stethoscope className="w-6 h-6" /></div>
                        <div>
                            <p className="text-[10px] font-bold text-slate-400 uppercase tracking-widest">Patolog√≠a Frecuente</p>
                            <h3 className="text-lg font-black text-slate-800 leading-tight">{stats.topPatologia}</h3>
                        </div>
                    </div>

                    {/* 6. PROMEDIO IMC */}
                    <div className="bg-white p-5 rounded-2xl shadow-sm border border-slate-100 flex items-center gap-4 hover:shadow-md transition">
                        <div className="p-3 bg-indigo-50 rounded-xl text-indigo-600"><Scale className="w-6 h-6" /></div>
                        <div>
                            <p className="text-[10px] font-bold text-slate-400 uppercase tracking-widest">IMC Promedio</p>
                            <h3 className="text-2xl font-black text-slate-800">{stats.promedioIMC}</h3>
                            <p className="text-[9px] text-indigo-400 font-bold">Perfil de Poblaci√≥n</p>
                        </div>
                    </div>

                </div>

                {/* BARRA DE T√çTULO DE TABLA */}
                <div className="flex items-center justify-between mb-4">
                    <h2 className="text-lg font-bold text-slate-700 flex items-center gap-2">
                        <FileText className="w-5 h-5 text-slate-400" /> Registro de Pacientes
                    </h2>
                    <span className="bg-slate-200 text-slate-600 px-2 py-1 rounded text-xs font-bold">Total: {pautas.length}</span>
                </div>

                {/* BUSCADOR */}
                <div className="bg-white p-4 rounded-2xl shadow-sm border border-slate-200 mb-6 flex gap-3 items-center focus-within:ring-2 focus-within:ring-blue-100 transition-all">
                    <Search className="w-5 h-5 text-slate-300" />
                    <input type="text" placeholder="Buscar paciente por nombre..." className="w-full outline-none text-slate-600 font-medium" value={busqueda} onChange={(e) => setBusqueda(e.target.value)} />
                </div>

                {/* TABLA INFERIOR */}
                <div className="bg-white rounded-2xl shadow-xl border border-slate-200 overflow-hidden">
                    <table className="w-full text-left">
                        <thead>
                            <tr className="bg-slate-50 border-b border-slate-200 text-[10px] font-black text-slate-400 uppercase tracking-widest">
                                <th className="p-5">Fecha / Hora</th>
                                <th className="p-5">Paciente & Estrategia</th>
                                <th className="p-5 text-center">Calor√≠as</th>
                                <th className="p-5 text-center">Objetivo</th>
                                <th className="p-5 text-right">Acciones</th>
                            </tr>
                        </thead>
                        <tbody className="divide-y divide-slate-100">
                            {pautasFiltradas.length > 0 ? (
                                pautasFiltradas.map((p) => {
                                    const f = new Date(p.fecha);
                                    return (
                                    <tr key={p.id} className="hover:bg-blue-50/30 transition-colors group">
                                        <td className="p-5">
                                            <div className="flex items-center gap-2 text-slate-700 font-bold text-sm">
                                                <Calendar className="w-3.5 h-3.5 text-blue-500" /> {f.toLocaleDateString('es-CL')}
                                            </div>
                                            <div className="text-[10px] text-slate-400 font-medium ml-5">{f.toLocaleTimeString('es-CL', {hour: '2-digit', minute:'2-digit'})} hrs</div>
                                        </td>
                                        <td className="p-5">
                                            <div className="font-extrabold text-slate-800 text-base uppercase tracking-tight">{p.paciente}</div>
                                            <div className="text-[11px] text-slate-500 font-semibold italic">{ESTRATEGIAS_LABELS[p.estrategia] || "Personalizada"}</div>
                                        </td>
                                        <td className="p-5 text-center">
                                            <span className="bg-slate-100 text-slate-700 px-3 py-1.5 rounded-lg text-xs font-black border border-slate-200">{Math.round(p.caloriasMeta || 0)} <span className="text-[9px] opacity-60">KCAL</span></span>
                                        </td>
                                        <td className="p-5 text-center">{getObjetivoBadge(p.objetivo)}</td>
                                        <td className="p-5 text-right">
                                            <div className="flex justify-end gap-1">
                                                <Link href={`/paciente/${p.paciente}`} className="p-2.5 text-purple-600 hover:bg-purple-100 rounded-xl transition-all" title="Ver Evoluci√≥n"><LineChart className="w-5 h-5" /></Link>
                                                <button onClick={() => cargarEnCalculadora(p)} className="p-2.5 text-blue-600 hover:bg-blue-100 rounded-xl transition-all" title="Ver Detalle / Editar"><FileText className="w-5 h-5" /></button>
                                                <button onClick={() => borrarPauta(p.id)} className="p-2.5 text-slate-300 hover:text-red-500 hover:bg-red-50 rounded-xl transition-all" title="Eliminar"><Trash2 className="w-5 h-5" /></button>
                                            </div>
                                        </td>
                                    </tr>
                                )})
                            ) : (
                                <tr><td colSpan={5} className="p-10 text-center text-slate-400 font-medium">No se encontraron pacientes...</td></tr>
                            )}
                        </tbody>
                    </table>
                </div>
            </div>
        </div>
    );
}
